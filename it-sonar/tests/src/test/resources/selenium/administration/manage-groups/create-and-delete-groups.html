create-and-delete-groups
open /sonar/sessions/new
type login admin
type password admin
clickAndWait commit
open /sonar/groups/index
store javascript{new Date().getTime()} groupId
type group_name ${groupId}
type group_description Description of ${groupId}
clickAndWait commit
assertVisible info
assertText group-${groupId} glob:*Description of ${groupId}*
clickAndWait delete-${groupId}
assertConfirmation Are you sure that you want to delete this group? Members will not be deleted.
assertVisible info
assertElementNotPresent group-${groupId}
