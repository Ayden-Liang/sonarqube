create-and-delete-groups
open /sonar/sessions/new
type login admin
type password admin
clickAndWait commit
open /sonar/groups/index
store javascript{new Date().getTime()} groupId
click link-create-group
waitForText group_form *Add new group*
type group_name ${groupId}
type group_description Description of ${groupId}
click name=commit
waitForText info *The new group is created.*
click delete-${groupId}
waitForText css=#confirm-form .modal-head *Delete group ${groupId}*
waitForText confirm-form *Are you sure that you want to delete this group?*Members will not be deleted.?*
click confirm-submit
waitForText info *Group is deleted.*
assertElementNotPresent group-${groupId}
